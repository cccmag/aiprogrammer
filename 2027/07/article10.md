# AI 基礎設施即程式碼

## 前言

AI 系統的基礎設施比傳統應用複雜一個數量級：GPU 驅動與 CUDA 版本管理、模型服務框架的配置、向量資料庫的索引策略、提示詞路由器的快取規則、以及跨多個 Kubernetes 叢集的資源排程。手動管理這些是不可行的——不僅效率低，而且容易出錯。

Infrastructure as Code（IaC）是唯一的可行路徑。2027 年的 AI IaC 已經從單純的 Terraform 配置演進為專為 AI 工作負載設計的宣告式框架。本文將展示如何使用 IaC 原則管理 AI 基礎設施。

## AI 基礎設施的核心元件

一個典型的 AI 生產環境包含以下元件：

```python
from dataclasses import dataclass, field
from typing import Optional

@dataclass
class ModelService:
    name: str
    model_name: str          # Hugging Face 或 MLflow 模型 URI
    model_version: str       # 版本或 "latest"
    framework: str           # triton, bentoml, tf-serving
    replicas: int = 2
    gpu_type: str = "A100"
    gpu_count: int = 1
    min_replicas: int = 1
    max_replicas: int = 10
    target_latency_ms: int = 500
    resources: dict = field(default_factory=lambda: {
        "cpu": "4", "memory": "16Gi"
    })
    autoscaling_metrics: list = field(default_factory=lambda: [
        "avg_gpu_utilization", "request_queue_depth"
    ])

@dataclass
class VectorStore:
    name: str
    engine: str = "chroma"   # chroma, qdrant, pinecone
    dimension: int = 1536
    index_type: str = "hnsw"
    distance_metric: str = "cosine"
    replicas: int = 2
    persistence: dict = field(default_factory=lambda: {
        "size": "100Gi",
        "snapshot_interval": "24h"
    })

@dataclass
class PromptRouter:
    name: str
    strategy: str = "latency_based"  # latency_based, random, weighted
    cache_enabled: bool = True
    cache_ttl_seconds: int = 3600
    rate_limit: int = 1000
    rate_limit_window: str = "1m"

@dataclass
class MonitoringStack:
    prometheus_retention: str = "30d"
    grafana_dashboards: list = field(default_factory=lambda: [
        "model_performance", "rag_quality", "agent_behavior"
    ])
    alert_rules: dict = field(default_factory=lambda: {
        "error_rate": {"threshold": 0.01, "window": "5m"},
        "p99_latency": {"threshold": 2000, "window": "5m"},
        "drift_detected": {"threshold": 0.2, "window": "1h"},
    })

@dataclass
class AIInfrastructure:
    project: str
    environment: str  # dev, staging, production
    model_services: list[ModelService] = field(default_factory=list)
    vector_stores: list[VectorStore] = field(default_factory=list)
    prompt_routers: list[PromptRouter] = field(default_factory=list)
    monitoring: MonitoringStack = field(default_factory=MonitoringStack)
```

## 宣告式基礎設施定義

```python
# 宣告完整的 AI 基礎設施
class AIInfrastructureDeclarer:
    def __init__(self, project: str, environment: str):
        self.infra = AIInfrastructure(
            project=project,
            environment=environment,
        )

    def add_llm_service(self, name: str, model: str,
                         version: str = "latest",
                         replicas: int = 2) -> "AIInfrastructureDeclarer":
        self.infra.model_services.append(ModelService(
            name=name,
            model_name=model,
            model_version=version,
            framework="triton",
            replicas=replicas,
            gpu_type="A100",
            target_latency_ms=300 if self.infra.environment == "production" else 1000,
        ))
        return self

    def add_embedding_service(self, name: str,
                               model: str = "text-embedding-3-large",
                               replicas: int = 2) -> "AIInfrastructureDeclarer":
        self.infra.model_services.append(ModelService(
            name=name,
            model_name=model,
            model_version="latest",
            framework="bentoml",
            replicas=replicas,
            gpu_type="T4",
            gpu_count=1,
            target_latency_ms=200,
        ))
        return self

    def add_vector_store(self, name: str,
                          engine: str = "chroma",
                          dimension: int = 1536) -> "AIInfrastructureDeclarer":
        self.infra.vector_stores.append(VectorStore(
            name=name,
            engine=engine,
            dimension=dimension,
        ))
        return self

    def add_prompt_router(self, name: str) -> "AIInfrastructureDeclarer":
        self.infra.prompt_routers.append(PromptRouter(name=name))
        return self

    def generate_terraform(self) -> str:
        """生成 Terraform 配置（示意）"""
        config = f"""
# AI Infrastructure — {self.infra.project} ({self.infra.environment})
# Generated by AIInfrastructureDeclarer

terraform {{
  required_providers {{
    kubernetes = {{ source = "hashicorp/kubernetes" }}
    helm = {{ source = "hashicorp/helm" }}
  }}
}}

provider "kubernetes" {{
  config_path = "~/.kube/config"
}}

# Model Services
"""
        for svc in self.infra.model_services:
            config += f"""
resource "kubernetes_deployment" "{svc.name}" {{
  metadata {{
    name = "{svc.name}"
    labels = {{
      app = "{svc.name}"
      environment = "{self.infra.environment}"
    }}
  }}
  spec {{
    replicas = {svc.replicas}
    selector {{
      match_labels = {{
        app = "{svc.name}"
      }}
    }}
    template {{
      metadata {{
        labels = {{
          app = "{svc.name}"
        }}
      }}
      spec {{
        container {{
          image = "nvcr.io/nvidia/tritonserver:24.12-py3"
          name = "{svc.name}"
          args = ["tritonserver", "--model-repository=/models"]
          resources {{
            limits = {{
              nvidia.com/gpu = {svc.gpu_count}
              cpu = "{svc.resources['cpu']}"
              memory = "{svc.resources['memory']}"
            }}
          }}
          env {{
            name = "MODEL_NAME"
            value = "{svc.model_name}"
          }}
        }}
      }}
    }}
  }}
}}

resource "kubernetes_service" "{svc.name}" {{
  metadata {{
    name = "{svc.name}-svc"
  }}
  spec {{
    selector = {{
      app = "{svc.name}"
    }}
    port {{
      port = 8000
      target_port = 8000
      name = "http"
    }}
    port {{
      port = 8001
      target_port = 8001
      name = "grpc"
    }}
  }}
}}
"""
        return config

    def generate_kubernetes_yaml(self) -> dict:
        """生成 Kubernetes YAML 配置"""
        configs = {}
        for svc in self.infra.model_services:
            configs[f"{svc.name}-deployment.yaml"] = f"""
apiVersion: apps/v1
kind: Deployment
metadata:
  name: {svc.name}
  namespace: ai-{self.infra.environment}
spec:
  replicas: {svc.replicas}
  selector:
    matchLabels:
      app: {svc.name}
  template:
    metadata:
      labels:
        app: {svc.name}
    spec:
      containers:
      - name: {svc.name}
        image: nvcr.io/nvidia/tritonserver:24.12-py3
        args: ["tritonserver", "--model-repository=/models"]
        resources:
          limits:
            nvidia.com/gpu: {svc.gpu_count}
            cpu: {svc.resources['cpu']}
            memory: {svc.resources['memory']}
        env:
        - name: MODEL_NAME
          value: {svc.model_name}
        - name: MODEL_VERSION
          value: {svc.model_version}
"""
        return configs

    def deploy(self, dry_run: bool = True) -> dict:
        """部署基礎設施（模擬）"""
        import json

        plan = {
            "project": self.infra.project,
            "environment": self.infra.environment,
            "services": [
                {
                    "name": s.name,
                    "type": "model_service",
                    "model": s.model_name,
                    "replicas": s.replicas,
                    "gpu": f"{s.gpu_type}x{s.gpu_count}",
                }
                for s in self.infra.model_services
            ],
            "vector_stores": [
                {
                    "name": v.name,
                    "engine": v.engine,
                    "dimension": v.dimension,
                }
                for v in self.infra.vector_stores
            ],
            "prompt_routers": [
                {"name": r.name, "strategy": r.strategy}
                for r in self.infra.prompt_routers
            ],
            "monitoring": {
                "alert_rules": self.infra.monitoring.alert_rules,
            },
        }

        if dry_run:
            print("=== 基礎設施部署計畫（預覽模式）===")
            print(json.dumps(plan, indent=2))
            return {"status": "preview", "plan": plan}

        # 實際部署邏輯（呼叫 Kubernetes API / Terraform）
        print(f"部署至 {self.infra.environment} 環境...")
        return {"status": "deployed", "plan": plan}
```

## 使用範例

```python
# 定義並部署一個 AI 基礎設施
def deploy_ai_infrastructure():
    # 宣告基礎設施
    infra = AIInfrastructureDeclarer("customer-support", "production")

    infra \
        .add_llm_service("main-llm", "gpt-4o-mini",
                         replicas=3, version="latest") \
        .add_embedding_service("embedder", "text-embedding-3-small",
                               replicas=2) \
        .add_vector_store("knowledge-base", engine="chroma",
                          dimension=1536) \
        .add_prompt_router("main-router")

    # 預覽
    plan = infra.deploy(dry_run=True)

    # 生成 K8s 配置
    k8s_configs = infra.generate_kubernetes_yaml()
    for name, yaml in k8s_configs.items():
        with open(f"deploy/{name}", "w") as f:
            f.write(yaml)

    print(f"已生成部署設定檔至 deploy/ 目錄")

    return plan
```

## GPU 資源管理

GPU 是 AI 基礎設施中最昂貴也最稀缺的資源。IaC 需要解決 GPU 排程問題：

```python
class GPUResourceManager:
    def __init__(self):
        self.cluster_gpus = {
            "a100-40gb": 8,
            "a100-80gb": 4,
            "h100": 2,
        }
        self.allocations: dict[str, dict] = {}

    def request_gpu(self, service_name: str, gpu_type: str,
                     count: int = 1) -> dict:
        """請求 GPU 分配"""
        available = self.cluster_gpus.get(gpu_type, 0) - \
            sum(
                a.get("gpu_type") == gpu_type and a.get("count", 0)
                for a in self.allocations.values()
            )

        if available < count:
            return {
                "status": "rejected",
                "reason": f"GPU {gpu_type} 不足: 需要 {count}，可用 {available}",
                "alternatives": [
                    gtype for gtype, total in self.cluster_gpus.items()
                    if total - sum(
                        a.get("gpu_type") == gtype and a.get("count", 0)
                        for a in self.allocations.values()
                    ) >= count
                ]
            }

        allocation = {
            "service": service_name,
            "gpu_type": gpu_type,
            "count": count,
            "allocated_at": datetime.now().isoformat(),
        }
        self.allocations[service_name] = allocation
        return {"status": "allocated", **allocation}

    def optimize_scheduling(self) -> list[dict]:
        """最佳化 GPU 排程"""
        suggestions = []
        # GPU 叢集利用率分析
        for gpu_type, total in self.cluster_gpus.items():
            allocated = sum(
                a.get("count", 0) for a in self.allocations.values()
                if a.get("gpu_type") == gpu_type
            )
            utilization = allocated / total * 100
            if utilization < 50:
                suggestions.append({
                    "gpu_type": gpu_type,
                    "utilization": f"{utilization:.0f}%",
                    "suggestion": "考慮縮減或共用 GPU",
                })
            elif utilization > 90:
                suggestions.append({
                    "gpu_type": gpu_type,
                    "utilization": f"{utilization:.0f}%",
                    "suggestion": "接近滿載，建議擴充",
                })
        return suggestions
```

## 端到端基礎設施部署管線

```python
# 基礎設施即程式碼的完整管線
class AIIaCPipeline:
    def __init__(self):
        self.stages = []

    def run(self, infra: AIInfrastructureDeclarer):
        print(f"=== 開始部署: {infra.infra.project}/{infra.infra.environment} ===")

        # 階段 1：生成配置
        print("[1/4] 生成基礎設施配置...")
        terraform_config = infra.generate_terraform()
        k8s_configs = infra.generate_kubernetes_yaml()

        # 階段 2：配置審查
        print("[2/4] 配置審查...")
        review = self._review_config(terraform_config)
        if not review["passed"]:
            print(f"審查失敗: {review['issues']}")
            return {"status": "failed", "stage": "review"}

        # 階段 3：部署
        print("[3/4] 部署基礎設施...")
        result = infra.deploy(dry_run=False)

        # 階段 4：驗證
        print("[4/4] 部署驗證...")
        verification = self._verify_deployment(infra)

        return {
            "status": "completed" if verification["passed"] else "degraded",
            "terraform_config": terraform_config,
            "deployment_result": result,
            "verification": verification,
        }

    def _review_config(self, config: str) -> dict:
        # 模擬配置審查
        return {"passed": True, "issues": []}

    def _verify_deployment(self, infra: AIInfrastructureDeclarer) -> dict:
        # 模擬部署驗證
        checks = []
        for svc in infra.infra.model_services:
            checks.append({
                "service": svc.name,
                "endpoint_reachable": True,
                "latency_ok": True,
                "gpu_allocated": True,
            })
        return {
            "passed": all(c["endpoint_reachable"] for c in checks),
            "checks": checks,
        }
```

## 實戰建議

1. **環境隔離**：使用獨立的 Kubernetes Namespace 或叢集區分 dev/staging/production
2. **秘密管理**：API Key、資料庫密碼使用 Kubernetes Secrets 或 Vault 管理，不要寫在配置中
3. **版本鎖定**：固定 GPU 驅動、CUDA 版本、模型服務框架版本，避免「在我機器上可以跑」
4. **基礎設施測試**：使用 Terratest 或類似工具測試基礎設施配置的有效性
5. **變更審計**：所有基礎設施變更都通過 Git 進行，保留完整變更歷史
